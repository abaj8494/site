** {{< leetcode easy checked >}} 1. Two Sum

#+BEGIN_SRC python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        table = dict()
        for i, num in enumerate(nums):
            if target - num in table.keys():
                return [i, table[target-num]]
            table[num] = i
        return []
#+END_SRC
